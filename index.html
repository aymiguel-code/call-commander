Haz clic en: Implementar → Nueva implementación * - Tipo: App web * - Ejecutar como: Yo (tu cuenta de Google) * - Quién puede acceder: Cualquier persona (Anyone) * * 5. Haz clic en "Implementar" y acepta los permisos. * * 6. Copia la URL generada (termina en /exec). * * 7. En Call Commander → ⚙️ → pega la URL → Guardar → Probar. * * NOTA: La hoja "Llamadas" y "Prospectos" se crean automáticamente * si no existen. Si ya tienes datos en la primera hoja, se usan como * fuente para "Cargar desde Sheets". */ // ===== CONFIGURACIÓN ===== const SPREADSHEET_ID = '1T_NkUSlIWLnNJ8CnTEnqog52bYmmuo2eZeqqP4xri1A'; const SHEET_LLAMADAS = 'Llamadas'; const SHEET_PROSPECTOS = 'Prospectos'; const SHEET_SEGUIMIENTO = '📋 SEGUIMIENTO'; // CRM principal de Miguel (65 contactos activos) const SHEET_DIRECTORIO = '📞 DIRECTORIO'; // Directorio crudos de contactos // ===== MAPA DE NORMALIZACIÓN DE COLUMNAS ===== // Traduce nombres de columna de cualquier hoja al formato estándar de Call Commander. // // Fuentes soportadas: // - Tab "Prospectos" : columnas en Title Case (ID, Nombre, Teléfono, Tipo...) // - Tab "📋 SEGUIMIENTO": columnas en MAYÚSCULAS con emoji (NOMBRE, ESTRATEGIA, ETAPA DEL EMBUDO...) // - Tab "📞 DIRECTORIO" : columnas en MAYÚSCULAS (NOMBRE, TELÉFONO, FUENTE / GRUPO, NOTAS) const COLUMN_MAP = { // Nombre 'NOMBRE': 'Nombre', 'NOMBRE COMPLETO': 'Nombre', // Teléfono 'TELÉFONO': 'Teléfono', 'TELEFONO': 'Teléfono', 'CELULAR': 'Teléfono', 'PHONE': 'Teléfono', // Tipo de contacto — SEGUIMIENTO usa ESTRATEGIA (Vida/Reclutamiento), DIRECTORIO usa FUENTE / GRUPO 'ESTRATEGIA': 'Tipo', // SEGUIMIENTO: Vida | Reclutamiento 'FUENTE / GRUPO': 'Tipo', // DIRECTORIO: Propios | Lyft | Diego Vega | ... 'GRUPO': 'Tipo', 'TIPO': 'Tipo', // Fuente de origen (solo SEGUIMIENTO) 'FUENTE': 'Fuente', // Propios | Keyla R | Fabian T | etc. // Estado y etapa del embudo (SEGUIMIENTO) 'ETAPA DEL EMBUDO': 'Etapa', // Primera Llamada | Seguimiento Post-Presentación | ... 'ESTADO ACTUAL': 'Estado', // Interesado | Volver a llamar | Revisando info | ... 'ESTADO': 'Estado', 'STATUS': 'Estado', // Seguimiento programado '⚡ PRÓXIMO SEGUIMIENTO': 'Callback', // Fecha del próximo contacto 'PRÓXIMO SEGUIMIENTO': 'Callback', // Sin emoji, por si acaso // Score MACHOE (SEGUIMIENTO) 'SCORE': 'Score', // 0–5 puntos de calificación // Métricas adicionales (SEGUIMIENTO) '# INTENTOS': 'Intentos', 'ASIGNADO A': 'AsignadoA', 'PÓLIZA': 'Poliza', // Notas y referidos 'NOTAS': 'Notas', 'NOTA': 'Notas', 'REFERIDO POR': 'Referido Por', 'REFERIDO': 'Referido Por', // Comunes 'ID': 'ID', 'PRIORIDAD': 'Prioridad' }; function normalizeCol(raw) { const upper = (raw || '').toString().trim().toUpperCase(); return COLUMN_MAP[upper] || raw; } // ===== MANEJADOR GET (lectura) ===== // Soporta JSONP: si se envía ?callback=nombreFuncion, la respuesta se envuelve // en nombreFuncion(json) para eludir la restricción CORS en navegadores web. function doGet(e) { try { const action = (e.parameter && e.parameter.action) || 'ping'; const callback = (e.parameter && e.parameter.callback) || null; let data; if (action === 'ping') { data = { success: true, message: 'Call Commander conectado correctamente ✅' }; } else if (action === 'getProspectos') { data = getProspectosData(); } else if (action === 'getCallbacks') { data = getCallbacksData(); } else if (action === 'logCall') { // Escritura vía GET+JSONP — evita el problema del redirect que pierde el body en POST data = logCallFromParams(e.parameter); } else { data = { success: true, message: 'API activa.', acciones: ['ping','getProspectos','getCallbacks','logCall'] }; } return makeResponse(data, callback); } catch (ex) { const cb = (e.parameter && e.parameter.callback) || null; return makeResponse({ success: false, error: 'doGet: ' + ex.message }, cb); } } // Devuelve TextOutput normal (JSON) o JSONP según si hay callback. function makeResponse(data, callback) { const json = JSON.stringify(data); if (callback) { // JSONP — el navegador ejecuta el script y llama la función definida en Call Commander return ContentService .createTextOutput(callback + '(' + json + ')') .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService .createTextOutput(json) .setMimeType(ContentService.MimeType.JSON); } // ===== MANEJADOR POST (escritura) ===== function doPost(e) { try { const data = JSON.parse(e.postData.contents); const action = data.action; if (action === 'logCall') { return jsonOk(logCall(data)); // logCall devuelve objeto plano → envolver en TextOutput } if (action === 'addProspecto') { return addProspecto(data); } if (action === 'updateProspecto') { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); actualizarEstadoProspecto(ss, data.nombre, data.estado, data.callback, data.notas); return jsonOk({ mensaje: 'Prospecto actualizado.' }); } return jsonError('Acción no reconocida: ' + action); } catch (ex) { return jsonError('doPost: ' + ex.message); } } // ===== REGISTRAR LLAMADA VÍA GET (JSONP) ===== // El redirect 302 de GAS convierte POST→GET y pierde el body. // Solución: aceptar logCall como acción GET con params en la URL. function logCallFromParams(p) { return logCall({ nombre: p.nombre || '', telefono: p.telefono || '', tipo: p.tipo || '', estado: p.estado || '', guion: p.guion || '', fechaCita: p.fechaCita || '', horaCita: p.horaCita || '', linkZoom: p.linkZoom || '', callback: p.callback || '', notas: p.notas || '' }); } // ===== REGISTRAR LLAMADA ===== function logCall(data) { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const sh = getOrCreateSheet(ss, SHEET_LLAMADAS, [ 'ID', 'Fecha', 'Hora', 'Nombre', 'Teléfono', 'Tipo', 'Estado', 'Guion', 'Fecha Cita', 'Hora Cita', 'Link Zoom', 'Callback', 'Notas' ]); const tz = Session.getScriptTimeZone(); const now = new Date(); const fecha = Utilities.formatDate(now, tz, 'yyyy-MM-dd'); const hora = Utilities.formatDate(now, tz, 'HH:mm'); const id = Utilities.getUuid().substring(0, 8).toUpperCase(); sh.appendRow([ id, fecha, hora, data.nombre || '', data.telefono || '', data.tipo || '', data.estado || '', data.guion || '', data.fechaCita || '', data.horaCita || '', data.linkZoom || '', data.callback || '', data.notas || '' ]); // Actualizar SEGUIMIENTO y Prospectos con el resultado de la llamada if (data.nombre) { try { actualizarEstadoProspecto(ss, data.nombre, data.estado, data.callback, data.notas); } catch (e) { Logger.log('actualizarEstadoProspecto error: ' + e.message); } } // Devuelve objeto plano para que logCallFromParams pueda ser usado en JSONP via doGet return { success: true, id: id, mensaje: 'Llamada registrada.' }; } // ===== AGREGAR PROSPECTO ===== function addProspecto(data) { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const sh = getOrCreateSheet(ss, SHEET_PROSPECTOS, [ 'ID', 'Nombre', 'Teléfono', 'Tipo', 'Referido Por', 'Estado', 'Prioridad', 'Fecha Agregado', 'Notas' ]); const tz = Session.getScriptTimeZone(); const hoy = Utilities.formatDate(new Date(), tz, 'yyyy-MM-dd'); const id = Utilities.getUuid().substring(0, 8).toUpperCase(); sh.appendRow([ id, data.nombre || '', data.telefono || '', data.tipo || '', data.referidoPor || '', 'Pendiente', data.prioridad || 'Normal', hoy, data.notas || '' ]); return jsonOk({ id: id, mensaje: 'Prospecto agregado.' }); } // ===== OBTENER PROSPECTOS (devuelve objeto plano, no TextOutput) ===== function getProspectosData() { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); // Prioridad de fuente: // 1. Tab "Prospectos" — generado por Call Commander, datos normalizados // 2. Tab "📋 SEGUIMIENTO" — CRM principal de Miguel: 65 contactos con etapa, score, estrategia // 3. Tab "📞 DIRECTORIO" — directorio crudo de 70 contactos // 4. Array vacío (instrucciones al usuario) let sh = ss.getSheetByName(SHEET_PROSPECTOS); let mode = 'prospectos'; if (!sh) { sh = ss.getSheetByName(SHEET_SEGUIMIENTO); mode = 'seguimiento'; } if (!sh) { sh = ss.getSheetByName(SHEET_DIRECTORIO); mode = 'directorio'; } if (!sh) return { success: true, data: [], total: 0, note: 'Crea el tab "Prospectos" en tu hoja o agrega contactos desde Call Commander.' }; const allData = sh.getDataRange().getValues(); if (allData.length < 2) return { success: true, data: [], total: 0, source: sh.getName() }; // Detectar fila de encabezado real — SEGUIMIENTO tiene 3 filas de título antes de los headers let headerRowIdx = 0; for (let i = 0; i < Math.min(allData.length, 5); i++) { const row = allData[i].join('|').toUpperCase(); if (row.includes('NOMBRE') && (row.includes('TELÉFONO') || row.includes('TELEFONO'))) { headerRowIdx = i; break; } } const headers = allData[headerRowIdx].map(h => normalizeCol(h)); let rows = allData.slice(headerRowIdx + 1) .filter(r => { // Para SEGUIMIENTO col 0 = "#" (número de fila), col 1 = Nombre // Para DIRECTORIO y Prospectos col 0 = Nombre / ID const a = r[0], b = r[1]; return (a !== '' && a !== null && a !== undefined) || (b !== '' && b !== null && b !== undefined); }) .map(r => { const obj = {}; headers.forEach((h, i) => { const val = r[i]; obj[h] = (val !== null && val !== undefined) ? val.toString() : ''; }); return obj; }); // Para SEGUIMIENTO: filtrar cerrados y ordenar por próximo seguimiento if (mode === 'seguimiento') { const CERRADAS = ['cerrado - ganado', 'cerrado - perdido']; rows = rows.filter(r => { const etapa = (r['Etapa'] || '').toLowerCase().trim(); return !CERRADAS.some(c => etapa.includes(c.replace(' - ', ''))) && !etapa.includes('cerrado'); }); // Ordenar: contactos con fecha de seguimiento primero (más urgentes), luego sin fecha rows.sort((a, b) => { const da = a['Callback'] ? new Date(a['Callback']) : new Date('9999-12-31'); const db = b['Callback'] ? new Date(b['Callback']) : new Date('9999-12-31'); return da - db; }); } return { success: true, data: rows, total: rows.length, source: sh.getName() }; } // ===== OBTENER CALLBACKS PENDIENTES (devuelve objeto plano) ===== function getCallbacksData() { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const sh = ss.getSheetByName(SHEET_LLAMADAS); if (!sh) return { success: true, data: [] }; const data = sh.getDataRange().getValues(); if (data.length < 2) return { success: true, data: [] }; const headers = data[0]; const now = new Date(); const pendientes = data.slice(1).filter(r => { if (!r[11]) return false; try { const d = new Date(r[11]); return !isNaN(d) && d >= now; } catch (e) { return false; } }).map(r => { const obj = {}; headers.forEach((h, i) => { obj[h] = r[i] ? r[i].toString() : ''; }); return obj; }); return { success: true, data: pendientes }; } // ===== ACTUALIZAR CONTACTO EN SEGUIMIENTO (y Prospectos si existe) ===== // Recibe: nombre, estado, callback (fecha próximo seguimiento), notas // Actualiza: ESTADO ACTUAL, ⚡ PRÓXIMO SEGUIMIENTO, # INTENTOS (+1), NOTAS function actualizarEstadoProspecto(ss, nombre, estado, callback, notas) { const nombreLower = nombre.toLowerCase().trim(); // 1. Actualizar SEGUIMIENTO (CRM principal) const shSeg = ss.getSheetByName(SHEET_SEGUIMIENTO); if (shSeg) { const allData = shSeg.getDataRange().getValues(); // Detectar fila de encabezado (SEGUIMIENTO tiene 3 filas de título antes) let headerRowIdx = 0; for (let i = 0; i < Math.min(allData.length, 5); i++) { const row = allData[i].join('|').toUpperCase(); if (row.includes('NOMBRE') && (row.includes('TELÉFONO') || row.includes('TELEFONO'))) { headerRowIdx = i; break; } } const headers = allData[headerRowIdx].map(h => (h || '').toString().trim().toUpperCase()); // Índices de columnas que vamos a actualizar const iNombre = headers.indexOf('NOMBRE'); const iEstado = headers.indexOf('ESTADO ACTUAL'); const iCallback = headers.findIndex(h => h.includes('PRÓXIMO SEGUIMIENTO') || h.includes('PROXIMO SEGUIMIENTO')); const iIntentos = headers.indexOf('# INTENTOS'); const iNotas = headers.indexOf('NOTAS'); // Buscar fila del contacto (empieza después del header) for (let i = headerRowIdx + 1; i < allData.length; i++) { const filaVal = ((iNombre >= 0 ? allData[i][iNombre] : allData[i][1]) || '').toString().toLowerCase().trim(); if (filaVal === nombreLower) { const rowNum = i + 1; // Sheets es 1-indexed if (iEstado >= 0 && estado) shSeg.getRange(rowNum, iEstado + 1).setValue(estado); if (iCallback >= 0 && callback) shSeg.getRange(rowNum, iCallback + 1).setValue(callback); if (iNotas >= 0 && notas) { const notaActual = (allData[i][iNotas] || '').toString().trim(); const tz = Session.getScriptTimeZone(); const hoy = Utilities.formatDate(new Date(), tz, 'MM/dd'); const nuevaNota = notaActual ? notaActual + ' | ' + hoy + ': ' + notas : hoy + ': ' + notas; shSeg.getRange(rowNum, iNotas + 1).setValue(nuevaNota); } if (iIntentos >= 0) { const actual = parseInt(allData[i][iIntentos] || '0', 10) || 0; shSeg.getRange(rowNum, iIntentos + 1).setValue(actual + 1); } SpreadsheetApp.flush(); break; } } } // 2. Actualizar tab Prospectos si existe (fallback / duplicado) const shPros = ss.getSheetByName(SHEET_PROSPECTOS); if (shPros) { const data = shPros.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { const fila = (data[i][1] || '').toString().toLowerCase().trim(); if (fila === nombreLower) { shPros.getRange(i + 1, 6).setValue(estado); // Col F = Estado break; } } } } // ===== CREAR HOJA SI NO EXISTE ===== function getOrCreateSheet(ss, nombre, headers) { let sh = ss.getSheetByName(nombre); if (!sh) { sh = ss.insertSheet(nombre); sh.appendRow(headers); // Formato del encabezado const rng = sh.getRange(1, 1, 1, headers.length); rng.setFontWeight('bold'); rng.setBackground('#0f172a'); rng.setFontColor('#00d46a'); sh.setFrozenRows(1); // Ancho automático de columnas for (let c = 1; c <= headers.length; c++) { sh.autoResizeColumn(c); } SpreadsheetApp.flush(); } return sh; } // ===== HELPERS JSON ===== function jsonOk(obj) { return ContentService .createTextOutput(JSON.stringify({ success: true, ...obj })) .setMimeType(ContentService.MimeType.JSON); } function jsonError(msg) { return ContentService .createTextOutput(JSON.stringify({ success: false, error: msg })) .setMimeType(ContentService.MimeType.JSON); }
